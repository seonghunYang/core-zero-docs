import { useState } from "react";

{/* Import CSS modules */}
import styles from '../components/counters.module.css'

export const Counter = () => {
  const [count, setCount] = useState(0);
  return (
    <div>
      <button
        className="bg-slate-200"
        onClick={() => setCount(count + 1)}
        // className={styles.counter}
      >
        Clicked {count} times
      </button>
    </div>
  );
};

<Counter />
